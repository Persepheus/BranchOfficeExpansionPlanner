	id = boep.500
	hide_window = yes
	trigger = {
		has_country_flag = boep_auto_establish_checked		
	}
	is_triggered_only = yes
	immediate = {
		set_variable = { which = boep_current_energy_cost_sum value = boep_auto_establish_min_energy }
		set_variable = { which = boep_current_influence_cost_sum value = boep_auto_establish_min_influence }
		change_variable = { which = boep_current_energy_cost_sum value = 850 }
		change_variable = { which = boep_current_influence_cost_sum value = 40 }
		if = {
			limit = { 
				resource_stockpile_compare = { resource = energy value >= boep_current_energy_cost_sum }
				resource_stockpile_compare = { resource = influence value >= boep_current_influence_cost_sum }
			}
		
			boep_calculate_data = yes

			boep_clear_values = yes
			if = {
				limit = { has_country_flag = boep_auto_establish_prio_cost_checked }
				boep_sort_and_establish = { KEY = boep_branch_office_influence_cost ASC = yes } 
			}
			else_if = {
				limit = { has_country_flag = boep_auto_establish_prio_income_checked }
				boep_sort_and_establish = { KEY = boep_current_branch_office_value ASC = no }			
			}
			else = {
				boep_sort_and_establish = { KEY = boep_branch_office_roi ASC = yes } 
			}
		}
	}
}

### Autobuild

country_event = {
	id = boep.700
	hide_window = yes
	trigger = {
		has_country_flag = boep_auto_build_checked
	}
	is_triggered_only = yes
	immediate = {

		boep_check_slots = yes
		
		every_galaxy_planet = {
			limit = {
				has_branch_office = root
				OR = {
					AND = {
						OR = { NOT = { is_variable_set = boep_queued_bo_building } check_variable = { which = boep_queued_bo_building value = 0 } }
						free_branch_office_building_slots >= 1
					}
					AND = {
						check_variable = { which = boep_queued_bo_building value = 1 }
						free_branch_office_building_slots >= 2
					}
					AND = {
						check_variable = { which = boep_queued_bo_building value = 2 }
						free_branch_office_building_slots >= 3
					}
					AND = {
						check_variable = { which = boep_queued_bo_building value = 3 }
						free_branch_office_building_slots >= 4
					}
				}
				
			}
			boep_count_buildings = yes

			boep_go_through_buildings = yes
		}
	}
}

						target = prev
						variable = {
							type = name
							localization = PLANET
							scope = prev
						}
					}
				}
			}
		}
	}
